CompilationUnit = Functions
Functions = Function | Functions Function

Function = Name LParen Parameters RParen Colon Expression Semicolon
         | Name LParen            RParen Colon Expression Semicolon
Parameters = Name | Parameters Comma Name

Expression = Primary | Expression Plus Primary
Primary = FunctionCall | Name | LParen Expression RParen
FunctionCall = Name LParen Arguments RParen | Name LParen RParen
Arguments = Expression | Arguments Comma Expression
